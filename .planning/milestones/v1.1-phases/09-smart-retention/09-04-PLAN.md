    ```python
    def graphiti_stale(scope: str = "auto") -> str:
        """Preview nodes eligible for TTL archiving.

        Returns TOON-encoded list of stale nodes with name, age (days), and score.
        Score 0.0 = most stale, 1.0 = healthiest.

        Args:
            scope: "auto" (default), "global", or "project"

        Returns:
            TOON-encoded stale node list or JSON for small result sets.
        """
        code, out, err = _run_graphiti(
            ["stale", "--format", "json", "--all"] + _scope_flags(scope)
        )
        if code != 0:
            return f"Error running graphiti stale: {err or out}"
        return _parse_json_or_raw(out, "stale")
    ```

    Key decisions:
    - Uses --format json --all flags so output is parseable JSON (all nodes, not capped at 25)

    - Uses stdlib logging (NOT structlog) — this is inside mcp_server/
