    Design:

    ```python
    class LLMError(Exception):
        """Raised when an LLM operation fails. Message names provider and URL."""

    @dataclass
    class LLMResponse:
        content: str
        provider: str
        model: str

    class LLMClient:
        """Single-provider LLM client. No fallback, no retry on other providers."""

        def __init__(self, config: Config):
            self._config = config
            self._provider = config.llm.provider  # "claude" | "ollama" | "openai"

        async def chat(self, messages: list[dict], **kwargs) -> LLMResponse: ...
        async def embed(self, texts: list[str]) -> list[list[float]]: ...

    def make_llm_client(config: Config) -> LLMClient:
        """Factory — construct LLMClient from config."""
        return LLMClient(config)
    ```

    Implement chat() via internal _dispatch:
    - provider="claude": call _chat_claude(messages)
      - shutil.which("claude") is None → raise LLMError("claude binary not found — install Claude Code CLI")
      - Build prompt from messages (role.upper() + ": " + content, joined by "\n\n")
      - asyncio.create_subprocess_exec("claude", "-p", prompt, "--output-format", "json", "--model", config.llm.model)
      - wait_for(proc.communicate(), timeout=120)
      - returncode != 0 → raise LLMError(f"claude -p failed (exit {returncode}): {stderr}")
      - Parse json.loads(stdout)["result"] → LLMResponse(content=result, provider="claude", model=config.llm.model)

    - provider="ollama": call _chat_ollama(messages)
      - url = config.llm.url or "http://localhost:11434"
      - headers = {"Authorization": f"Bearer {config.llm.api_key}"} if config.llm.api_key else {}
      - POST {url}/api/chat with {"model": config.llm.model, "messages": messages, "stream": False}
      - timeout=30.0 (httpx.AsyncClient)
      - httpx.ConnectError / httpx.TimeoutException → raise LLMError(f"ollama unreachable at {url}: {e}")
      - response.status_code != 200 → raise LLMError(f"ollama error {response.status_code} at {url}")
      - Parse response.json()["message"]["content"]

    - provider="openai": call _chat_openai(messages)
      - url = config.llm.url (must not be empty — raise LLMError("openai provider requires url in [llm] config") if empty)
      - headers = {"Authorization": f"Bearer {config.llm.api_key}", "Content-Type": "application/json"}
      - POST {url}/chat/completions with {"model": config.llm.model, "messages": messages}
      - timeout=30.0
      - httpx.ConnectError / httpx.TimeoutException → raise LLMError(f"openai-compatible provider unreachable at {url}: {e}")
      - response.status_code != 200 → raise LLMError(f"openai-compatible provider error {response.status_code} at {url}")
      - Parse response.json()["choices"][0]["message"]["content"]

    Implement embed():
    - config.embeddings is None → raise LLMError("embeddings not configured — add [embeddings] section to config.toml")
    - embeddings.provider="ollama": POST {embeddings.url}/api/embed with {"model": embeddings.model, "input": texts}
      - parse response.json()["embeddings"]
    - embeddings.provider="openai": POST {embeddings.url}/embeddings with {"model": embeddings.model, "input": texts}
      - headers with Bearer api_key if set
      - parse [item["embedding"] for item in response.json()["data"]]
    - Wrap all httpx errors → LLMError naming the URL

    Use structlog.get_logger(__name__) for debug logging (no logging import).
