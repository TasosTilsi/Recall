Structure:

```python
"""End-to-end smoke test: recall init → search → mcp serve.

Marked @pytest.mark.integration — requires a working LLM config and the
recall CLI to be installed (pip install -e . or pipx install recall-kg).
Run with: pytest tests/test_e2e_smoke.py -v -m integration
Skip in CI: pytest tests/ -m "not integration"
"""
import json
import os
import subprocess
import sys
import time
import pytest
from pathlib import Path

RECALL_BIN = sys.executable.replace("python", "recall") if "python" in sys.executable else "recall"

def _recall(*args, cwd=None, env=None, timeout=60, input=None):
    """Run `recall <args>` as a subprocess, return (returncode, stdout, stderr)."""
    cmd = [RECALL_BIN, *args]
    result = subprocess.run(
        cmd,
        cwd=cwd,
        env=env,
        capture_output=True,
        text=True,
        timeout=timeout,
        input=input,
    )
    return result.returncode, result.stdout, result.stderr


@pytest.mark.integration
def test_recall_help():
    """recall --help lists the six-command surface."""
    rc, out, err = _recall("--help")
    assert rc == 0, f"recall --help failed:\n{err}"
    # Six commands: init, sync, search, health, config, ui
    for cmd in ("init", "sync", "search", "health", "config", "ui"):
        assert cmd in out, f"'{cmd}' missing from --help output:\n{out}"


@pytest.mark.integration
def test_init_search_mcp_smoke(tmp_path):
    """Full workflow: init indexes current repo, search returns results, mcp serve starts."""
    # Set up isolated config pointing to tmp db
    config_dir = tmp_path / ".recall"
    config_dir.mkdir()
    db_path = config_dir / "recall.db"
    config_toml = config_dir / "config.toml"
    config_toml.write_text(
        f"""
[llm]
provider = "claude"
model = "claude-haiku-4-5-20251001"
url = ""
api_key = ""

[embeddings]
provider = "ollama"
model = "nomic-embed-text"
url = "http://localhost:11434"
api_key = ""

[db]
path = "{db_path}"
"""
    )

    # Phase 27's load_config() resolves via Path.home() / ".recall" / "config.toml".
    # Redirect HOME so the subprocess finds our temp config without env-var support.
    env = {**os.environ, "HOME": str(tmp_path)}
    repo_root = Path(__file__).parent.parent  # project root

    # Step 1: recall init
    rc, out, err = _recall("init", cwd=str(repo_root), env=env, timeout=120)
    assert rc == 0, f"recall init failed (exit {rc}):\nSTDOUT: {out}\nSTDERR: {err}"
    assert db_path.exists(), "recall init did not create the database file"

    # Step 2: recall search "fix"
    rc, out, err = _recall("search", "fix", cwd=str(repo_root), env=env, timeout=30)
    assert rc == 0, f"recall search failed (exit {rc}):\nSTDOUT: {out}\nSTDERR: {err}"
    # Results may be empty if repo has no indexed data yet, but command must not error

    # Step 3: recall mcp serve — send initialize, read response, kill
    proc = subprocess.Popen(
        [RECALL_BIN, "mcp", "serve"],
        cwd=str(repo_root),
        env=env,
        stdin=subprocess.PIPE,
        stdout=subprocess.PIPE,
        stderr=subprocess.PIPE,
        text=True,
    )
    initialize_msg = json.dumps({
        "jsonrpc": "2.0",
        "id": 1,
        "method": "initialize",
        "params": {
            "protocolVersion": "2024-11-05",
            "capabilities": {},
            "clientInfo": {"name": "smoke-test", "version": "0"},
        },
    }) + "\n"
    try:
        proc.stdin.write(initialize_msg)
        proc.stdin.flush()
        time.sleep(2)  # give server time to respond
        proc.terminate()
        stdout, stderr = proc.communicate(timeout=10)
    except subprocess.TimeoutExpired:
        proc.kill()
        proc.communicate()
        pytest.fail("recall mcp serve did not exit within 10s after terminate")

    assert proc.returncode in (0, -15), (
        f"mcp serve exited unexpectedly (exit {proc.returncode}):\n{stderr}"
    )
```

Key implementation notes:
- Find the `recall` binary by inspecting the current venv's bin dir: use `shutil.which("recall")` as fallback, then `sys.executable`-based path
- Replace the `RECALL_BIN` assignment at module level with:
  ```python
  import shutil
  RECALL_BIN = shutil.which("recall") or str(Path(sys.executable).parent / "recall")
  ```
- Config isolation uses `HOME=tmp_path` — Phase 27's `load_config()` in `src/config.py` resolves
  `Path.home() / ".recall" / "config.toml"` by default; redirecting HOME is the correct mechanism.
  Do NOT use a `RECALL_CONFIG` env var (Phase 27 does not support it).
- Mark the whole module with `pytestmark = pytest.mark.integration` for easy skip.
