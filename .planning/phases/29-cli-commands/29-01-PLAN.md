    **Step 1: Create src/cli/commands/sync_cmd.py**

    New file implementing the `recall sync` command. This is an incremental indexer wrapper:

    ```python
    """Sync command — incremental git history indexer."""
    import typer
    from src.cli.output import console, print_success, print_error, print_warning
    from src.cli.utils import EXIT_SUCCESS, EXIT_ERROR

    def sync_command() -> None:
        """Index new git commits since last sync.

        Incremental — only processes commits not yet indexed.
        Automatically runs a full init if no database exists yet.

        Examples:
            recall sync
        """
        try:
            from pathlib import Path
            import subprocess, sys, shutil

            # Find git repo root (walk up from cwd)
            root = _find_git_root()
            if root is None:
                print_error(
                    "Not in a git repository. Cannot sync git history.",
                    suggestion="Navigate to a git repository and try again"
                )
                raise typer.Exit(EXIT_ERROR)

            # Check if DB exists — if not, run full init instead
            db_path = root / ".recall" / "recall.db"
            if not db_path.exists():
                console.print("[dim]No database found — running full init...[/dim]")
                _run_full_index(root)
                raise typer.Exit(EXIT_SUCCESS)

            # Incremental sync
            from src.indexer import GitIndexer
            indexer = GitIndexer(project_root=root)
            with console.status("Syncing new commits..."):
                result = indexer.run(full=False)

            _print_index_result(result)

        except typer.Exit:
            raise
        except Exception as e:
            print_error(f"Sync failed: {str(e)}")
            raise typer.Exit(EXIT_ERROR)
    ```

    Also define helpers `_find_git_root() -> Path | None` (walks up from Path.cwd() looking for `.git/`) and `_run_full_index(root: Path)` (calls GitIndexer with reset_full then run(full=True)), and `_print_index_result(result: dict)` (prints commits_processed, elapsed_seconds, entity_names_sample in same style as init_cmd.py).

    ```python
    """Health command — check LLM provider and database connectivity."""
    import typer
    from src.cli.output import console, print_success, print_error

    def health_command() -> None:
        """Check LLM provider and database status.

        Verifies connectivity to the configured LLM provider and embeddings
        service. Completes within 5 seconds.

        Examples:
            recall health
        """
        try:
            from src.llm.health import check_health
            from src.config import load_config

            with console.status("Checking health..."):
                result = check_health(timeout=5.0)

            # Provider line
            llm_status = "OK" if result["llm_ok"] else "UNREACHABLE"
            console.print(f"Provider : {result['provider']}  model={result['model']}  [{llm_status}]")

            # Embeddings line
            if result.get("embeddings_model"):
                emb_status = "OK" if result["embeddings_ok"] else "UNREACHABLE"
                console.print(f"Embeddings: {result['embeddings_model']}  [{emb_status}]")
            else:
                console.print("Embeddings: not configured")

            # Error detail (if any)
            if result.get("error"):
                console.print(f"[dim]Error detail: {result['error']}[/dim]")

            # Exit code: 0 if LLM reachable, 1 if not
            if not result["llm_ok"]:
                raise typer.Exit(code=1)

        except typer.Exit:
            raise
        except Exception as e:
            print_error(f"Health check failed: {str(e)}")
            raise typer.Exit(code=1)
    ```

    - Output format: one line for provider/LLM status, one line for embeddings status (or "not configured")
    - Exit 0 if LLM is reachable, exit 1 if not
